class Flag < ActiveRecord::Base
  belongs_to :challenge, inverse_of: :flags
  has_many :answers, inverse_of: :flag

  structure do
    point       200, validates: :presence
    flag        'FLAG_kogasa', validates: :presence
    timestamps
  end

  def point=(value)
    value = Integer(value)
    self[:point] = value if value >= 0
  end
end
