def suspend!(period = 900)
  raise ArgumentError, "`period' must not be nil." unless period
  case period
  when Integer, Float
    self.suspended_until = Time.now + period
  when Time
    self.suspended_until = period
  else
    raise TypeError, "`period' should be Integer or Time."
  end
  save!
end
